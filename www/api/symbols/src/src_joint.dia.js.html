 70 /**
 71  * Abstract object of all diagram elements.
 72  * This object is never used directly, instead, specific diagram elements inherits from it.
 73  * Allows easy creation of new specific diagram elements preserving all features that Joint library and Joint.dia plugin offer.
 74  * <h3>Wrapper</h3>
 75  *  All custom elements must have a wrapper set. Wrapper is the key object that Joint library counts with. 
 76  *  There cannot be any element without a wrapper. Usually it is an object which wraps all the subelements 
 77  *  that a specific diagram element contains. The wrapper must be set in init method.
 78  *  To set a wrapper, use setWrapper(aWrapper) method. The single parameter to the method is a Raphaël vector object. 
 79  *  Later on, you can access this object using wrapper property.
 80  * <h3>Inner</h3>
 81  *  Inner objects are subelements of an element. Although they are optional, they are commonly used. To add a subelement 
 82  *  to the element, use addInner(anInner) method. It takes a Raphaël vector object as an argument. All inner objects are 
 83  *  placed to an array that you can access using inner property.
 84  * <h3><i>init</i> method</h3>
 85  *  The <i>init</i> method has to be part of every element you create. It takes all element options as an argument, 
 86  *  sets wrapper and adds inners.
 87  * <h3><i>joint</i> method</h3>
 88  *  If you have specific elements, in which connections are not controlled by wrapper, you can implement your own joint method.
 89  * <h3><i>zoom</i> method</h3>
 90  *  As Joint.dia library does not know how your specific element should behave after scaling, you can use zoom method to implement 
 91  *  the desired behaviour.
 92  * @name Element
 93  * @memberOf Joint.dia
 94  * @constructor
 95  * @example
 96 var mydia = Joint.dia.mydia = {};
 97 var Element = Joint.dia.Element;
 98 
 99 mydia.MyElement = Element.extend({
100   // init method must be always presented
101   init: function(properties){
102     var p = this.properties;
103     // parameters processing
104     p.position = properties.position;
105     p.radius = properties.radius || 30;
106     // every element must have a wrapper
107     this.setWrapper(this.paper.circle(p.position.x, p.position.y, p.radius));
108     // optional inner elements 
109     this.addInner(this.paper.text(p.position.x, p.position.y, "my element"));
110   }
111 });

116   position: {x: 50, y: 50},

122 /**
123  * Copies all the properties to the first argument from the following arguments.
124  * All the properties will be overwritten by the properties from the following
125  * arguments. Inherited properties are ignored.
126  * @private
127  */
128 var Mixin = Joint.Mixin = function() {
129     var target = arguments[0];
130     for (var i = 1, l = arguments.length; i < l; i++){
131         var extension = arguments[i];
132         for (var key in extension){
133             if (!extension.hasOwnProperty(key)){
134 		continue;
135 	    }
136             var copy = extension[key];
137             if (copy === target[key]){
138 		continue;
139 	    }
140             // copying super with the name base if it does'nt has one already
141             if (typeof copy == "function" && typeof target[key] == "function" && !copy.base){
142 		copy.base = target[key];
143 	    }
144             target[key] = copy;
145         }
146     }
147     return target;
148 };

150 /**
151  * Copies all properties to the first argument from the following
152  * arguments only in case if they don't exists in the first argument.
153  * All the function propererties in the first argument will get
154  * additional property base pointing to the extenders same named
155  * property function's call method.
156  * @example
157  * // usage of base
158  * Bar.extend({
159  * // function should have name
160  * foo: function foo(digit) {
161  * return foo.base(this, parseInt(digit))
162  * }
163  * });
164  * @private
165  */
166 var Supplement = Joint.Supplement = function() {
167     var target = arguments[0];
168     for (var i = 1, l = arguments.length; i < l; i++){
169         var extension = arguments[i];
170         for (var key in extension) {
171             var copy = extension[key];
172             if (copy === target[key]){
173 		continue;
174 	    }
175             // copying super with the name base if it does'nt has one already
176             if (typeof copy == "function" && typeof target[key] == "function" && !target[key].base){
177 		target[key].base = copy;
178 	    }
179             // target doesn't has propery that is owned by extension copying it
180             if (!target.hasOwnProperty(key) && extension.hasOwnProperty(key)){
181 		target[key] = copy;
182 	    }
183         }
184     }
185     return target;
186 };

675 /**
676  * Document mousemove event.
677  * @private
678  */
679 Element.mouseMove = function(e){
680     e = e || window.event;
681     // object dragging
682     if (dia._currentDrag){
683 	if (dia._currentDrag._opt.ghosting)	// if ghosting, move ghost
684 	    dia._currentDrag.ghost.translate(e.clientX - dia._currentDrag.dx, e.clientY - dia._currentDrag.dy);
685 	else	// otherwise, move the whole shape
686 	    dia._currentDrag.translate(e.clientX - dia._currentDrag.dx, e.clientY - dia._currentDrag.dy);
687 
688 	dia._currentDrag.dx = e.clientX;
689 	dia._currentDrag.dy = e.clientY;
690     }
691 
692     // object zooming
693     if (dia._currentZoom){
694 	var 
695 	dx = e.clientX - dia._currentZoom.dx,
696 	dy = e.clientY - dia._currentZoom.dy;
697 	
698 	dia._currentZoom.dWidth -= dx;
699 	dia._currentZoom.dHeight -= dy;
700 	// correction
701 	if (dia._currentZoom.dWidth < 1) dia._currentZoom.dWidth = 1;
702 	if (dia._currentZoom.dHeight < 1) dia._currentZoom.dHeight = 1;
703
